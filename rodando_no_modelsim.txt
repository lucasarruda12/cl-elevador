# Na janela Transcript do ModelSim, digite:
vlib work
vcom door.vhd
vcom udc.vhd
vcom llel.vhd
vcom in_controller.vhd

vsim work.in_controller

add wave -radix binary clk
add wave -radix binary int_floor_request
add wave -radix binary ext_floor_request
add wave -radix unsigned fr
add wave -radix binary intension
add wave -radix binary request_vector
add wave -radix binary /in_controller/up_int
add wave -radix binary /in_controller/dn_int
add wave -radix binary /in_controller/op_int
add wave -radix binary /in_controller/cl_int
add wave -radix binary /in_controller/dr_int
add wave -radix unsigned /in_controller/fr_int
add wave -radix integer /in_controller/calls_count

# Cria um clock de 100ns de período (50ns alto, 50ns baixo)
force clk 0 0ns, 1 50ns -repeat 100ns

# Inicializa as entradas em zero
force int_floor_request 00000000000000000000000000000000
force ext_floor_request 00000000000000000000000000000000

# Roda para inicializar
run 200ns

# Ativa o pedido
force int_floor_request 00000000000000000000000000000100
run 100ns

# IMPORTANTE: Zera o pedido (senão fica re-adicionando)
force int_floor_request 00000000000000000000000000000000

# Deixa o elevador trabalhar
run 2000ns
